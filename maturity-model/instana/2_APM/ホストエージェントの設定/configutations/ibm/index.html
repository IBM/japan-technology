  enabled: true
  local:
    poll_rate_configuration:
      os_poll_rate: 15  # OS メトリクス収集間隔（秒）
      db2_poll_rate: 15  # DB2 メトリクス収集間隔（秒）
      disabled_components:
        components: 'HISTORY_LOGS,JOB_QUEUES'  # 無効化するコンポーネント
      custom_poll_rate:
        poll_rate_1:
          polling_rate: 30
          components: 'DISK_STATUS,MESSAGE_QUEUE'
          events: 'DISK_ERROR,QUEUE_FULL'
    user_specification:
      activeJobs:
        jobs: '*/QUSER/QZDASOINIT,*/QSYS/*'
        event:
          identicalJobs:
            jobName/user: 'QZDASOINIT/*'
            threshold: 5

    設定項目の説明: - os_poll_rate: OSレベルのメトリクス収集間隔 - db2_poll_rate: DB2メトリクス収集間隔 - disabled_components: 監視から除外するコンポーネント - custom_poll_rate: コンポーネント別のカスタム収集間隔

    Apache Kafka

    📚 公式ドキュメント: Apache Kafka

    1
2
3

    com.instana.plugin.kafka:
  jmxUsername: 'jmx_user'
  jmxPassword: 'jmx_password'
  jmxPort: '1099'
  topicsRegex: '.*'
  poll_rate: 1

    設定項目の説明: - jmxUsername: JMX認証用ユーザー名（JMX認証が有効な場合） - jmxPassword: JMX認証用パスワード - jmxPort: JMXポート番号（デフォルト: 1099） - topicsRegex: 監視対象トピックの正規表現パターン（デフォルト: '.*' = 全て） - poll_rate: メトリクス収集間隔（秒単位、デフォルト: 1秒）

    # Kafkaサーバーの設定（server.properties）
# JMXポートの設定
export JMX_PORT=1099

# JMX認証を有効化する場合
export KAFKA_JMX_OPTS="-Dcom.sun.management.jmxremote \
  -Dcom.sun.management.jmxremote.authenticate=true \
  -Dcom.sun.management.jmxremote.ssl=false \
  -Dcom.sun.management.jmxremote.password.file=/path/to/jmxremote.password \
  -Dcom.sun.management.jmxremote.access.file=/path/to/jmxremote.access"

    com.instana.plugin.kafka:
  clusters:
    - name: 'kafka-prod'
      jmxPort: '1099'
      jmxUsername: 'monitoring'
      jmxPassword: 'prod_password'
      topicsRegex: '.*'
    - name: 'kafka-staging'
      jmxPort: '2099'
      topicsRegex: '.*'

    取得方法: - jmxPort: Kafkaサーバーの起動スクリプトまたはserver.propertiesで確認 - jmxUsername/Password: JMX認証ファイル（jmxremote.password）から取得 - topicsRegex: 監視要件に応じて設定（全トピック監視は負荷が高い場合あり）
